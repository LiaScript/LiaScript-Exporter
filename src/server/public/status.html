${window.i18n.t('status.no_job_id')}

` } else { loadStatus() refreshInterval = setInterval(loadStatus, 3000) } async function loadStatus() { try { const response = await fetch(`/api/job/${jobId}`) if (!response.ok) { throw new Error(window.i18n.t('status.job_not_found')) } const data = await response.json() displayStatus(data) // Stop refreshing if job is completed or failed if (data.job.status === 'completed' || data.job.status === 'failed') { clearInterval(refreshInterval) } } catch (error) { document.getElementById('statusContent').innerHTML = `

${window.i18n.t('status.error_loading')} ${error.message}

` clearInterval(refreshInterval) } } function displayStatus(data) { const { job, queuePosition } = data const statusClass = `status-${job.status}` const t = window.i18n.t let statusText = { queued: t('status.queued'), processing: t('status.processing'), completed: t('status.completed'), failed: t('status.failed'), }[job.status] let progressPercent = 0 if (job.status === 'processing') progressPercent = 50 if (job.status === 'completed') progressPercent = 100 const sourceType = job.source.type === 'upload' ? t('status.source_upload') : t('status.source_git') const locale = window.i18n.getCurrentLanguage() === 'en' ? ` : ''}

${job.startedAt ? `

${t('status.started_at')}: ${new Date(job.startedAt).toLocaleString(locale)}

${t('status.completed_at')}: ${new Date(job.completedAt).toLocaleString(locale)}

` : ''}
